import java.util.Scanner;
import java.util.*;

public class A3 {
	
	public static void main(String[] args){

		// declare variables
		double bmi;
		double weight;
		double height;

		// end variable decleration

		// Create scanner object

		Scanner scn = new Scanner(System.in);
		System.out.println("Welcome to my program!\nPlease enter your weight: ");
		weight = scn.nextDouble();
		System.out.println("Please enter your height: ");
		height = scn.nextDouble();

		bmi = ((weight * 703) / (height * height));

		System.out.println("Your body mass index (BMI) is: " + bmi);

		if(bmi <= 25 && bmi >= 18.5){

			System.out.println("Your weight is optimal.");
		} else if(bmi < 18.5 ){

			System.out.println("Your weight is under-weight. Time to bulk.");
		} else{

			System.out.println("Your weight is over-weight. Time to cut.");
		}


	}
}
